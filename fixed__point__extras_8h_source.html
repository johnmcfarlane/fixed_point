49  constexpr Rep sqrt_solve3(
50  Rep n,
51  Rep bit,
52  Rep result)
53  {
54  return (bit!=Rep{0})
55  ? (n>=result+bit)
56  ? sqrt_solve3<Rep>(
57  static_cast<Rep>(n-(result+bit)),
58  bit >> 2,

62  }
